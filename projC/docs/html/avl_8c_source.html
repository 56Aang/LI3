310 static NODO node_insert(NODO node, Valor info, Estrutura estrutura) {
311  int balance;
312 
313  if(node != NULL) {
314 
315  if (strcmp(info,node->string) < 0)
316  node->left = node_insert(node->left, info, estrutura);
317  else if(strcmp(info,node->string) > 0)
318  node->right = node_insert(node->right, info, estrutura);
319  else node->cont = estrutura;
320 
321  /* Atualiza os pesos */
322  node->height = max(height(node->left), height(node->right)) + 1;
323 
324  /* Varifica o balanceamento */
325  balance = getBalance(node);
326 
327  /* Left Left Case */
328  if (balance > 1 && strcmp(info,node->left->string) < 0) return rightRotate(node);
329 
330  /* Right Right Case */
331  if (balance < -1 && strcmp(info,node->right->string) > 0) return leftRotate(node);
332 
333  /* Left Right Case */
334  if (balance > 1 && strcmp(info,node->left->string) > 0) {
335  node->left = leftRotate(node->left);
336  return rightRotate(node);
337  }
338  /* Right Left Case */
339  if (balance < -1 && strcmp(info, node->right->string) < 0) {
340  node->right = rightRotate(node->right);
341  return leftRotate(node);
342  }
343 
344  } else node = newNode(node,info,estrutura);
345 
346  return node;
347 }
